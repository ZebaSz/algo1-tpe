\begin{problema}{crearC}{posG, posC: (\ent, \ent)}{Campo}
\requiere[puntosValidos]{0 \leq \prm{posG} \land 0 \leq \sgd{posG} \land 0 \leq \prm{posC} \land 0 \leq \sgd{posC}}
\requiere[puntosDistintos]{posG \neq posC}
\requiere[aMenosDe100]{distancia(posG, posC) \leq 100}
\requiere{distancia((0,0), posG) \leq 100}
\asegura{enRango(dimensiones(res), prm(posG), sgd(posG)) \\ \land ( contenido(res, prm(posG), sgd(posG)) == Granero)}
\asegura{enRango(dimensiones(res), prm(posC), sgd(posC)) \\ \land ( contenido(res, prm(posC), sgd(posC)) == Casa)}
\end{problema}

\begin{problema}{dimensionesC}{c: Campo}{(Ancho, Largo))}
\asegura{\res == dimensiones(c)}
\end{problema}

\begin{problema}{contenidoC}{c: Campo, i, j: \ent}{Parcela}
\requiere {enRango(dimensiones(c),i,j)}
\asegura{\res == contenido(c, i, j)}
\end{problema}
